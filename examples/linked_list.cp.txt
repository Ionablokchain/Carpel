Într-o arenă, pointerii înainte și înapoi (prev/next) sunt pur și simplu ref Node. Modificarea se face prin referințe exclusive (mut Node) atunci când avem nevoie de scriere, fără celule de interior.

// linked_list.cp
struct Node {
    value: i32,
    prev: Option<ref Node>,
    next: Option<ref Node>,
}

fn main() {
    let buffer: [u8; 1024] = [0; 1024];
    arena list_arena: Arena(buffer) {
        // Creăm head și tail ca referințe mutabile (exclusive)
        let mut head: mut Node = list_arena.alloc_mut(
            Node { value: 1, prev: None, next: None }
        );
        let mut tail: mut Node = list_arena.alloc_mut(
            Node { value: 3, prev: None, next: None }
        );

        // Legăm head <-> tail
        head.next = Some(tail);
        tail.prev = Some(head);

        // Inserăm un nod între ele
        let mut mid: mut Node = list_arena.alloc_mut(
            Node { value: 2, prev: None, next: None }
        );
        mid.prev = Some(head);
        mid.next = Some(tail);
        head.next = Some(mid);
        tail.prev = Some(mid);

        // Parcurgere cu referințe partajate (ref)
        let mut current: Option<ref Node> = Some(head);
        while let Some(node) = current {
            print!("{} ", node.value);
            current = node.next;
        }
        println!(); // 1 2 3
    }
}
